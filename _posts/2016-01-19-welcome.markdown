public ActionResult Rating()
{             
    var MonthRating = from u in dataContext.Users   
        where u.Likes.Count(l => l.Moderated &&

                            l.CreatedAt.Month == DateTime.Today.Month) > 0
        select u;

    return View(MonthRating);
}
